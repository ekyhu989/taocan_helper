### 数据结构

```typescript
interface Product {
  id: string;           // 唯一标识，使用uuid或时间戳
  name: string;         // 商品名称
  price: number;        // 单价
  unit: string;         // 单位
  category: string;     // 分类（食品/生活用品）
  category_tag: string; // 品类标签（食品/日用品/文体用品/其它节日礼品）
  scenes: Scene[];      // 适用场景数组
  is832: boolean;       // 832平台标识
  description?: string; // 描述（可选）
}

type Scene = 'holiday' | 'activity' | 'care';
```
