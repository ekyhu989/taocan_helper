```javascript
// historyStorage.ts
export function loadHistory(): HistoryRecord[] {
  const stored = localStorage.getItem('taocang_history');
  return stored ? JSON.parse(stored) : [];
}

export function saveHistory(record: HistoryRecord): void {
  const history = loadHistory();
  history.unshift(record);  // 新记录添加到开头
  if (history.length > 50) {
    history.pop();  // 超过 50 条，删除最旧
  }
  localStorage.setItem('taocang_history', JSON.stringify(history));
}

// ✅ 历史记录上限 50 条
// ✅ 使用 unshift 保证最新在前
// ✅ 超出限制自动清理
```

```javascript
// historyStorage.ts
export function loadHistory(): HistoryRecord[] {
  try {
    const stored = localStorage.getItem('taocang_history');
    if (!stored) return [];
    
    // ✅ JSON 格式验证
    const data = JSON.parse(stored);
    if (!Array.isArray(data)) {
      console.warn('历史数据格式错误，重置为空数组');
      return [];
    }
    return data;
  } catch (err) {
    // ✅ 异常处理
    console.warn('读取历史记录失败:', err);
    return [];
  }
}

export function saveHistory(record: HistoryRecord): void {
  try {
    const history = loadHistory();
    history.unshift(record);
    if (history.length > 50) {
      history.pop();  // ✅ 限制 50 条
    }

  } catch (err) {

      alert('存储空间不足，请导出历史记录释放空间');
    } else {
      console.error('保存历史记录失败:', err);
    }
  }
}
```

### 5.2 缺陷分析

**BUG-V1.6-4-001**: 恢复默认商品库缺少明确风险提示
- **发现用例**: TC-V1.6-4-007
- **问题描述**: 用户点击"恢复默认商品库"时，仅提示"确定恢复默认商品库？"，未明确说明"此操作将覆盖所有自定义商品"
- **影响范围**: 用户可能误操作丢失自定义商品数据
- **修复建议**: 
  ```javascript
  if (window.confirm('⚠️ 警告：此操作将覆盖所有自定义商品，恢复为默认商品库。确定继续吗？')) {
    resetToDefault();
  }
  ```
- **优先级**: P3（轻微）- 不影响核心功能，但影响用户体验

**BUG-V1.6-7-001**: 移动端输入框聚焦可能被键盘遮挡
- **发现用例**: TC-V1.6-7-004
- **问题描述**: 在移动端设备上，当键盘弹出时，底部输入框可能被键盘遮挡，用户无法看到输入内容
- **影响范围**: 移动端用户体验
- **修复建议**:
  ```javascript
  <input
    onFocus={(e) => {
      setTimeout(() => {
        e.target.scrollIntoView({ behavior: 'smooth', block: 'center' });
      }, 300);
    }}
  />
  ```
- **优先级**: P3（轻微）- 不影响核心功能，但影响用户体验
