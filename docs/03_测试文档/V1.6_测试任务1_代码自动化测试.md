## 三、测试用例要求

### 3.1 政策版本校验测试

```typescript
// 测试用例示例
describe('usePolicyVersion', () => {
  test('未确认版本时返回false', () => {
    localStorage.removeItem('policy_confirmed_version');
    expect(checkVersion()).toBe(false);
  });
  
  test('确认版本后返回true', () => {
    confirmVersion();
    expect(checkVersion()).toBe(true);
  });
  
  test('版本号不匹配时返回false', () => {
    localStorage.setItem('policy_confirmed_version', '2025-01-01');
    expect(checkVersion()).toBe(false);
  });
});
```

### 3.2 合规验证测试

```typescript
describe('complianceValidator', () => {
  test('人均500元触发黄色预警', () => {
    const result = validatePerCapitaBudget(500);

  });
  
  test('人均2000元触发橙色预警', () => {
    const result = validatePerCapitaBudget(2000);

  });
  
  test('832占比<30%触发红色预警', () => {

    expect(result.level).toBe('red');
  });
  
  test('数据锁定后不可修改', () => {
    lockData('admin');

  });
});
```

## 七、开发注意事项

1. **Mock依赖**：LocalStorage、CloudBase等需要Mock

4. **测试隔离**：每个测试用例独立，不依赖其他测试
