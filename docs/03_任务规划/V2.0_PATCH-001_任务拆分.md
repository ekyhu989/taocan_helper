## 任务1：组合目标价调整（GLM）

### 任务信息

- **负责人**: GLM（算法专家）
- **预估工时**: 4小时
- **优先级**: P0

### 需求描述
将组合算法目标从「人均基础预算」改为「人均总价上限」。

### 具体改动

#### 1.1 修改目标计算逻辑

**文件**: `src/utils/schemeGenerator.ts`

**当前逻辑**:
```typescript
// 以人均基础预算为目标
targetPrice = perCapitaBudget; // 如200元
```

**新逻辑**:
```typescript
// 计算人均总价上限
calculatePerCapitaLimit(
  perCapitaBudget: number,
  fundingSource: 'union' | 'other'
): number {
  // 工会经费: 预算 / 0.8 = 上限
  // 其他资金: 预算 / 0.9 = 上限
  const coefficient = fundingSource === 'union' ? 0.8 : 0.9;
  return Math.round(perCapitaBudget / coefficient);
}

// 目标函数: min(|组合总价 - 人均上限|), 且组合总价 ≤ 人均上限
targetPrice = perCapitaLimit; // 如250元
```

#### 1.2 修改组合算法

```typescript
// 在 generatePerCapitaScheme 方法中
function generatePerCapitaScheme(
  products: Product[],
  perCapitaBudget: number,
  fundingSource: 'union' | 'other',
  personCount: number
): Scheme {
  // 1. 计算人均上限
  const perCapitaLimit = calculatePerCapitaLimit(perCapitaBudget, fundingSource);
  
  // 2. 计算总价上限
  const totalLimit = perCapitaLimit * personCount;
  
  // 3. 组合算法目标：尽量接近totalLimit，但不超过
  const combination = findBestCombination(products, {
    target: totalLimit,
    max: totalLimit,
    min: totalLimit * 0.9 // 允许90%-100%区间
  });
  
  return combination;
}
```

- [ ] 生成的组合总价在240-250元区间（尽量接近250）
- [ ] 其他资金（9折）计算正确

#### 3.1 新增三方案生成方法

**文件**: `src/utils/schemeGenerator.ts`

```typescript
export interface SchemeSet {
  balanced: Scheme;      // 均衡推荐
  costEffective: Scheme; // 高性价比
  highQuality: Scheme;   // 高品质甄选
}

export function generateThreeSchemes(
  products: Product[],
  params: SchemeParams
): SchemeSet {
  // 1. 筛选832商品（占比要求）
  const products832 = products.filter(p => p.is832);
  
  // 2. 生成三个方案
  return {
    balanced: generateBalancedScheme(products, params),
    costEffective: generateCostEffectiveScheme(products832, params),
    highQuality: generateHighQualityScheme(products832, params)
  };
}

// 均衡推荐：isRecommended优先，832排前
function generateBalancedScheme(
  products: Product[],
  params: SchemeParams
): Scheme {
  const sortedProducts = [...products].sort((a, b) => {
    // isRecommended优先
    if (a.isRecommended && !b.isRecommended) return -1;
    if (!a.isRecommended && b.isRecommended) return 1;
    // 832商品优先
    if (a.is832 && !b.is832) return -1;
    if (!a.is832 && b.is832) return 1;
    return 0;
  });
  
  return generateScheme(sortedProducts, params);
}

// 高性价比：costPerformanceTag='高性价比'优先，倾向多件套
function generateCostEffectiveScheme(
  products832: Product[],
  params: SchemeParams
): Scheme {
  const sortedProducts = [...products832].sort((a, b) => {
    // 高性价比标签优先
    if (a.costPerformanceTag && !b.costPerformanceTag) return -1;
    if (!a.costPerformanceTag && b.costPerformanceTag) return 1;
    // 低价优先（高性价比倾向多件套）
    return a.price - b.price;
  });
  
  // 倾向多件套：选择更多数量的低价商品
  return generateSchemeWithPreference(sortedProducts, params, {
    preferMoreItems: true
  });
}

// 高品质甄选：qualityTag='高品质'优先，倾向单品/两件套
function generateHighQualityScheme(
  products832: Product[],
  params: SchemeParams
): Scheme {
  const sortedProducts = [...products832].sort((a, b) => {
    // 高品质标签优先
    if (a.qualityTag && !b.qualityTag) return -1;
    if (!a.qualityTag && b.qualityTag) return 1;
    // 高价优先（高品质通常价格较高）
    return b.price - a.price;
  });
  
  // 倾向单品/两件套：选择少量高价商品
  return generateSchemeWithPreference(sortedProducts, params, {
    preferMoreItems: false
  });
}
```

#### 3.2 832占比检查

```typescript
function check832Ratio(scheme: Scheme): { 
  ratio: number; 
  passed: boolean;
  warning?: string;
} {
  const totalAmount = scheme.totalAmount;
  const amount832 = scheme.items
    .filter(item => item.product.is832)
    .reduce((sum, item) => sum + item.product.price * item.quantity, 0);
  
  const ratio = (amount832 / totalAmount) * 100;
  
  return {
    ratio,
    passed: ratio >= 30,
    warning: ratio < 30 ? `832占比${ratio.toFixed(1)}%，低于30%政策红线` : undefined
  };
}
```

- [ ] 高品质优先qualityTag商品，倾向单品/两件套
- [ ] 三个方案都检查832占比，不达标时给出警告

#### 4.1 创建滑动卡片组件

**文件**: `src/components/SchemeCardSlider.tsx`

```typescript
import React, { useRef, useState } from 'react';
import { SchemeSet } from '../utils/schemeGenerator';

interface SchemeCardSliderProps {
  schemes: SchemeSet;
  onSchemeSelect?: (schemeType: 'balanced' | 'costEffective' | 'highQuality') => void;
}

export const SchemeCardSlider: React.FC<SchemeCardSliderProps> = ({
  schemes,
  onSchemeSelect
}) => {
  const scrollRef = useRef<HTMLDivElement>(null);
  const [activeIndex, setActiveIndex] = useState(0);
  
  const schemeList = [
    { type: 'balanced', name: '均衡推荐', scheme: schemes.balanced },
    { type: 'costEffective', name: '高性价比', scheme: schemes.costEffective },
    { type: 'highQuality', name: '高品质甄选', scheme: schemes.highQuality }
  ] as const;
  
  const handleScroll = () => {
    if (scrollRef.current) {
      const scrollLeft = scrollRef.current.scrollLeft;
      const cardWidth = scrollRef.current.offsetWidth * 0.85;
      const newIndex = Math.round(scrollLeft / cardWidth);
      setActiveIndex(newIndex);
    }
  };
  
  const scrollToCard = (index: number) => {
    if (scrollRef.current) {
      const cardWidth = scrollRef.current.offsetWidth * 0.85;
      scrollRef.current.scrollTo({
        left: index * cardWidth,
        behavior: 'smooth'
      });
    }
  };
  
  return (
    <div className="w-full">
      {/* 滑动卡片容器 */}
      <div
        ref={scrollRef}
        onScroll={handleScroll}
        className="flex overflow-x-auto snap-x snap-mandatory scrollbar-hide"
        style={{ scrollbarWidth: 'none', msOverflowStyle: 'none' }}
      >
        {schemeList.map((item, index) => (
          <div
            key={item.type}
            className="flex-shrink-0 w-[85%] md:w-[350px] snap-center px-2"
          >
            <SchemeCard
              type={item.type}
              name={item.name}
              scheme={item.scheme}
              isActive={index === activeIndex}
              onClick={() => onSchemeSelect?.(item.type)}
            />
          </div>
        ))}
      </div>
      
      {/* 指示器 */}
      <div className="flex justify-center gap-2 mt-4">
        {schemeList.map((_, index) => (
          <button
            key={index}
            onClick={() => scrollToCard(index)}
            className={`w-2 h-2 rounded-full transition-colors ${
              index === activeIndex ? 'bg-blue-600' : 'bg-gray-300'
            }`}
          />
        ))}
      </div>
      
      {/* 桌面端箭头按钮 */}
      <div className="hidden md:flex justify-center gap-4 mt-4">
        <button
          onClick={() => scrollToCard(Math.max(0, activeIndex - 1))}
          disabled={activeIndex === 0}
          className="px-4 py-2 bg-gray-100 rounded-lg disabled:opacity-50"
        >
          ← 上一个
        </button>
        <button
          onClick={() => scrollToCard(Math.min(2, activeIndex + 1))}
          disabled={activeIndex === 2}
          className="px-4 py-2 bg-gray-100 rounded-lg disabled:opacity-50"
        >
          下一个 →
        </button>
      </div>
    </div>
  );
};

// 单个方案卡片组件
interface SchemeCardProps {
  type: string;
  name: string;
  scheme: Scheme;
  isActive: boolean;
  onClick: () => void;
}

const SchemeCard: React.FC<SchemeCardProps> = ({
  type,
  name,
  scheme,
  isActive,
  onClick
}) => {
  // 计算832占比
  const totalAmount = scheme.totalAmount;
  const amount832 = scheme.items
    .filter(item => item.product.is832)
    .reduce((sum, item) => sum + item.product.price * item.quantity, 0);
  const ratio832 = (amount832 / totalAmount) * 100;
  
  // 计算品类数和商品件数
  const categoryCount = new Set(scheme.items.map(item => item.product.category)).size;
  const totalItems = scheme.items.reduce((sum, item) => sum + item.quantity, 0);
  
  return (
    <div
      onClick={onClick}
      className={`bg-white rounded-xl shadow-lg p-6 cursor-pointer transition-all ${
        isActive ? 'ring-2 ring-blue-500' : ''
      }`}
    >
      {/* 方案标签 */}
      <div className="flex items-center justify-between mb-4">
        <span className={`px-3 py-1 rounded-full text-sm font-medium ${
          type === 'balanced' ? 'bg-blue-100 text-blue-700' :
          type === 'costEffective' ? 'bg-green-100 text-green-700' :
          'bg-purple-100 text-purple-700'
        }`}>
          {name}
        </span>
        {ratio832 < 30 && (
          <span className="text-xs text-red-500">832占比不足</span>
        )}
      </div>
      
      {/* 核心数据 */}
      <div className="space-y-3">
        <div className="flex justify-between">
          <span className="text-gray-600">人均总价</span>
          <span className="text-xl font-bold text-gray-900">
            ¥{(scheme.totalAmount / scheme.personCount).toFixed(0)}
          </span>
        </div>
        
        <div className="flex justify-between">
          <span className="text-gray-600">832占比</span>
          <span className={`font-medium ${
            ratio832 >= 30 ? 'text-green-600' : 'text-red-500'
          }`}>
            {ratio832.toFixed(1)}%
          </span>
        </div>
        
        <div className="flex justify-between">
          <span className="text-gray-600">品类数</span>
          <span className="font-medium">{categoryCount}类</span>
        </div>
        
        <div className="flex justify-between">
          <span className="text-gray-600">商品件数</span>
          <span className="font-medium">{totalItems}件</span>
        </div>
      </div>
      
      {/* 商品预览 */}
      <div className="mt-4 pt-4 border-t border-gray-100">
        <p className="text-sm text-gray-500 mb-2">包含商品:</p>
        <div className="flex flex-wrap gap-1">
          {scheme.items.slice(0, 3).map((item, idx) => (
            <span key={idx} className="text-xs bg-gray-100 px-2 py-1 rounded">
              {item.product.name}×{item.quantity}
            </span>
          ))}
          {scheme.items.length > 3 && (
            <span className="text-xs text-gray-500">+{scheme.items.length - 3}</span>
          )}
        </div>
      </div>
    </div>
  );
};
```
