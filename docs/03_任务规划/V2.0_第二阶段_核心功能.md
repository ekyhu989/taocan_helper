#### 具体工作
1. **核心算法实现**:
   ```typescript
   // utils/algorithm/priceConstraint.ts
   interface PriceConstraintParams {
     totalBudget: number;
     peopleCount: number;
     fundSource: 'union' | 'other';
   }

   interface PriceConstraintResult {
     perCapitaBudget: number;      // 人均预算
     maxPerCapitaPrice: number;    // 人均价格上限
     discountRate: number;         // 折扣系数
     priceRange: {                 // 商品价格区间
       min: number;
       max: number;
     };
     calculationDetails: {         // 计算明细
       formula: string;
       steps: string[];
       policyReference: string;
     };
   }

   export function calculatePriceConstraint(
     params: PriceConstraintParams
   ): PriceConstraintResult {
     const { totalBudget, peopleCount, fundSource } = params;
     
     // 人均预算
     const perCapitaBudget = Math.floor(totalBudget / peopleCount);
     
     // 折扣系数：工会经费8折，其他资金9折
     const discountRate = fundSource === 'union' ? 0.8 : 0.9;
     
     // 价格上限
     const maxPerCapitaPrice = Math.ceil(perCapitaBudget / discountRate);
     
     // 价格区间：50% ~ 100%上限
     const priceRange = {
       min: Math.floor(maxPerCapitaPrice * 0.5),
       max: maxPerCapitaPrice
     };
     
     return {
       perCapitaBudget,
       maxPerCapitaPrice,
       discountRate,
       priceRange,
       calculationDetails: {
         formula: `上限 = 预算(${perCapitaBudget}) ÷ 折扣系数(${discountRate})`,
         steps: [
           `步骤1: 人均预算 = ${totalBudget} ÷ ${peopleCount} = ${perCapitaBudget}元`,
           `步骤2: 折扣系数 = ${fundSource === 'union' ? '0.8 (工会经费8折)' : '0.9 (其他资金9折)'}`,
           `步骤3: 价格上限 = ${perCapitaBudget} ÷ ${discountRate} = ${maxPerCapitaPrice}元`
         ],
         policyReference: '根据《工会经费管理办法》第X条规定...'
       }
     };
   }
   ```

   ```typescript
   // utils/compliance/calculator.ts
   export function calculate832Ratio(
     items: SchemeItem[],
     method: 'amount' | 'quantity'
   ): number {
     if (method === 'amount') {
       // 按金额计算
       const totalAmount = items.reduce((sum, item) => 
         sum + item.price * item.quantity, 0);
       const amount832 = items
         .filter(item => item.is832Platform)
         .reduce((sum, item) => sum + item.price * item.quantity, 0);
       return (amount832 / totalAmount) * 100;
     } else {
       // 按数量计算
       const totalQuantity = items.reduce((sum, item) => sum + item.quantity, 0);
       const quantity832 = items
         .filter(item => item.is832Platform)
         .reduce((sum, item) => sum + item.quantity, 0);
       return (quantity832 / totalQuantity) * 100;
     }
   }
   ```

   ```typescript
   // utils/annualData.ts
   export function getAnnualDataKey(key: string, year: number): string {
     return `${key}_${year}`;
   }

   export function saveAnnualData<T>(
     key: string, 
     year: number, 
     data: T
   ): void {
     const storageKey = getAnnualDataKey(key, year);
     localStorage.setItem(storageKey, JSON.stringify(data));
   }

   export function getAnnualData<T>(key: string, year: number): T | null {
     const storageKey = getAnnualDataKey(key, year);
     const data = localStorage.getItem(storageKey);
     return data ? JSON.parse(data) : null;
   }
   ```
