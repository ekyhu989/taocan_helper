### 任务4: 数据转换和传递

**文件**: `src/pages/desktop/DocumentView.tsx`

**添加数据转换逻辑**:

```typescript
// 将方案数据转换为公文模板需要的格式
const transformData = (
  solutionData: SolutionData,
  reportFormData: ReportFormData
): DocumentTemplateProps => {
  const { formData, productList, schemes, selectedSchemeIndex } = solutionData;
  
  // 获取当前选中的方案
  const currentScheme = schemes[selectedSchemeIndex];
  
  // 计算折扣率
  const discountRate = formData.fundingSource === '工会经费' ? 0.8 : 0.9;
  
  // 转换商品数据
  const products = currentScheme.items.map(item => ({
    name: item.product.name,
    quantity: item.quantity,
    unit: item.product.unit || '份',
    price: item.product.price,
    is832: item.product.is832Platform || false
  }));
  
  // 计算832占比
  const totalAmount = currentScheme.totalAmount;
  const amount832 = currentScheme.items
    .filter(i => i.product.is832Platform)
    .reduce((sum, i) => sum + i.product.price * i.quantity, 0);
  const platform832Rate = totalAmount > 0 ? amount832 / totalAmount : 0;
  
  return {
    unitName: reportFormData.unitName || formData.unitName || '某某单位',
    department: reportFormData.department || '工会',
    applicant: reportFormData.applicant || '',
    year: reportFormData.year || new Date().getFullYear(),
    personCount: formData.headCount,
    totalBudget: formData.totalBudget,
    perCapitaStandard: formData.perCapitaStandard,
    perCapitaActual: currentScheme.totalAmount / formData.headCount,
    fundingSource: formData.fundingSource,
    discountRate,
    products,
    platform832Rate,
    isCompliant: platform832Rate >= 0.3
  };
};

// 使用转换后的数据渲染
const documentProps = transformData(solutionData, reportFormData);

return (
  <div className="h-screen flex flex-col bg-gray-200">
    {/* 顶部栏 */}
    <div className="bg-white border-b border-gray-200 px-6 py-4 flex items-center justify-between">
      <div className="flex items-center gap-4">
        <h1 className="text-xl font-bold text-gray-900">公文排版视图</h1>
      </div>
      <div className="flex items-center gap-3">
        <button
          onClick={handlePrint}
          className="px-4 py-2 bg-gray-100 text-gray-700 font-medium rounded-lg hover:bg-gray-200 transition-colors flex items-center gap-2"
        >
          🖨️ 打印
        </button>
        <button
          onClick={handleExport}
          className="px-4 py-2 bg-blue-600 text-white font-medium rounded-lg hover:bg-blue-700 transition-colors flex items-center gap-2"
        >
          📥 导出
        </button>
      </div>
    </div>

    {/* 文档内容 */}
    <div className="flex-1 overflow-y-auto p-8">
      <DocumentTemplate {...documentProps} />
    </div>
  </div>
);
```
