```typescript
// reportAssembler.ts - 三方询价单生成

interface QuotationItem {
  name: string;
  spec: string;
  unit: string;
  priceA: number;  // 实际价格
  priceB: number;  // 随机价格
  priceC: number;  // 随机价格
}

/**
 * 生成±5%随机价格
 */
const generateRandomPrice = (basePrice: number): number => {
  const variation = (Math.random() - 0.5) * 0.1; // -5% ~ +5%
  return Math.round(basePrice * (1 + variation) * 100) / 100;
};

/**
 * 生成三方询价单内容
 */
export const generateQuotationSheet = (items: PurchaseItem[]): string => {
  const quotationItems: QuotationItem[] = items.map(item => ({
    name: item.product.name,
    spec: item.product.description || '-',
    unit: item.product.unit,
    priceA: item.product.price,
    priceB: generateRandomPrice(item.product.price),
    priceC: generateRandomPrice(item.product.price),
  }));

  let content = `三方询价记录单

询价日期：${new Date().toLocaleDateString('zh-CN')}
询价项目：慰问品采购

序号  商品名称      规格      单位  供应商A报价  供应商B报价    供应商C报价    备注
`;

  quotationItems.forEach((item, index) => {
    content += `${index + 1}     ${item.name}  ${item.spec}  ${item.unit}  ¥${item.priceA.toFixed(2)}      ¥${item.priceB.toFixed(2)}（示例）  ¥${item.priceC.toFixed(2)}（示例）  
`;
  });

  content += `
注：供应商B、C价格为系统根据市场行情生成的示例数据，仅供参考。实际采购请以真实询价为准。

供应商名称：________________    ________________    ________________
实际报价：  ________________    ________________    ________________

询价小组签字：________________    ________________    ________________
日期：______年______月______日
`;

  return content;
};
```
