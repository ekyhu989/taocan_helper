## 三、技术实现要点

### 3.1 回收站逻辑

**文件**：`src/utils/productStorage.ts`

```typescript
// 软删除
export const softDeleteProduct = (id: string) => {
  const product = getProduct(id);
  product.isDeleted = true;
  product.deletedAt = new Date().toISOString();
  product.restoreDeadline = addDays(new Date(), 7).toISOString();
  saveProduct(product);
};

// 恢复商品
export const restoreProduct = (id: string) => {
  const product = getProduct(id);
  product.isDeleted = false;
  delete product.deletedAt;
  delete product.restoreDeadline;
  saveProduct(product);
};

// 清理过期商品（每天检查一次）
export const cleanExpiredProducts = () => {
  const products = getAllProducts();
  const expired = products.filter(p => {
    if (!p.isDeleted || !p.restoreDeadline) return false;
    return new Date() > new Date(p.restoreDeadline);
  });
  expired.forEach(p => permanentlyDeleteProduct(p.id));
};
```
