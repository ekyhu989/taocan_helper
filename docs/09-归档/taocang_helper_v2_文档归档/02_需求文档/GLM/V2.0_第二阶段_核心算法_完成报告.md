## 🚀 使用示例

### 示例1：完整方案生成流水线

```typescript
import { 
  calculatePriceConstraint,
  filterEligibleProducts,
  generateOptimalCombination,
  runComplianceEngine
} from '../utils/algorithm';

async function generateFullScheme(basicInfo, allProducts) {
  const { totalBudget, peopleCount, fundSource } = basicInfo;
  
  // Step 1: 计算价格约束
  const priceResult = calculatePriceConstraint({
    totalBudget,
    peopleCount,
    fundSource
  });
  
  console.log(`人均预算: ${priceResult.perCapitaBudget}元`);
  console.log(`单价上限: ${priceResult.maxPerCapitaPrice}元`);
  console.log(`推荐价格区间: ${priceResult.priceRange.min}-${priceResult.priceRange.max}元`);
  
  // Step 2: 筛选匹配商品
  const matchingResult = filterEligibleProducts({
    products: allProducts,
    priceRange: priceResult.priceRange,
    prefer832: true,
    scene: basicInfo.scene || 'holiday',
    maxItems: 15
  });
  
  console.log(`筛选出${matchingResult.statistics.finalCount}件商品`);
  console.log(`其中832平台商品${matchingResult.statistics.platform832Count}件`);
  
  // Step 3: 生成最优组合
  if (matchingResult.eligibleProducts.length > 0) {
    const combination = generateOptimalCombination({
      products: matchingResult.eligibleProducts,
      targetBudget: totalBudget,
      peopleCount: peopleCount,
      minCategories: 3
    }, { target832Ratio: 0.35 });
    
    console.log(`组合总价: ${combination.totalPrice}元`);
    console.log(`人均价格: ${combination.perCapitaPrice}元`);
    console.log(`预算偏差: ${(combination.budgetDeviation * 100).toFixed(2)}%`);
    
    // Step 4: 合规检查
    const complianceResult = runComplianceEngine({
      items: combination.items,
      totalBudget: combination.totalPrice,
      peopleCount: peopleCount,
      fundSource: fundSource
    });
    
    console.log(`是否合规: ${complianceResult.isCompliant ? '✅ 是' : '❌ 否'}`);
    console.log(`合规评分: ${complianceResult.complianceScore}/100`);
    console.log(complianceResult.summaryReport);
    
    return {
      priceConstraint: priceResult,
      matching: matchingResult,
      combination: combination,
      compliance: complianceResult
    };
  }
}
```

import { quickComplianceCheck } from '../utils/algorithm';

function checkSchemeCompliance(schemeItems, budget, count) {
  const result = quickComplianceCheck({
    items: schemeItems,
    totalBudget: budget,
    peopleCount: count
  });
  
  if (!result.isCompliant) {
    alert(`⚠️ 方案存在合规风险！\n\n${result.message}\n\n建议：${result.suggestions.join('\n')}`);
    return false;
  }
  
  return true;
}
