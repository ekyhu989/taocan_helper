## P1: Page4合规测算自动化测试

### ComplianceCalculator.test.jsx

```jsx
import { render, screen, fireEvent } from '@testing-library/react';
import ComplianceCalculator from '../src/components/ComplianceCalculator';

// Mock localStorage
const mockLocalStorage = {
  getItem: jest.fn(),
  setItem: jest.fn(),
};
Object.defineProperty(window, 'localStorage', {
  value: mockLocalStorage,
});

describe('ComplianceCalculator - P1 Tests', () => {
  beforeEach(() => {
    mockLocalStorage.getItem.mockReturnValue(JSON.stringify({ headCount: 100 }));
  });

  // TC-P1-001: 年度进度测算-初始状态
  test('should display 0% completion with red progress bar initially', () => {
    render(<ComplianceCalculator />);
    
    const totalInput = screen.getByLabelText(/年度总预算/i);
    fireEvent.change(totalInput, { target: { value: '100000' } });
    
    expect(screen.getByText('0.0%')).toBeInTheDocument();
    expect(screen.getByText(/未达标，建议还需采购 30000 元/i)).toBeInTheDocument();
  });

  // TC-P1-002: 832完成率30%临界
  test('should show yellow progress bar at 30% completion', () => {
    render(<ComplianceCalculator />);
    
    fireEvent.change(screen.getByLabelText(/年度总预算/i), { target: { value: '100000' } });
    fireEvent.change(screen.getByLabelText(/已完成采购金额/i), { target: { value: '30000' } });
    
    expect(screen.getByText('30.0%')).toBeInTheDocument();
    expect(screen.getByText(/进行中，已达到最低要求/i)).toBeInTheDocument();
  });

  // TC-P1-004 ~ TC-P1-006: 2000元年度上限校验
  describe('2000元年度上限校验', () => {
    test('should not show warning when per capita is 1800 yuan', () => {
      render(<ComplianceCalculator />);
      
      fireEvent.change(screen.getByLabelText(/年度总预算/i), { target: { value: '180000' } });
      
      expect(screen.queryByText(/超过 2000 元上限/i)).not.toBeInTheDocument();
    });

    test('should not show warning at 2000 yuan threshold', () => {
      render(<ComplianceCalculator />);
      
      fireEvent.change(screen.getByLabelText(/年度总预算/i), { target: { value: '200000' } });
      
      expect(screen.queryByText(/超过 2000 元上限/i)).not.toBeInTheDocument();
    });

    test('should show orange warning when per capita exceeds 2000 yuan', () => {
      render(<ComplianceCalculator />);
      
      fireEvent.change(screen.getByLabelText(/年度总预算/i), { target: { value: '250000' } });
      
      expect(screen.getByText(/合规警告/i)).toBeInTheDocument();
      expect(screen.getByText(/年度人均预算 2500 元，超过 2000 元上限/i)).toBeInTheDocument();
    });
  });

  // TC-P1-008: 数据自动读取
  test('should auto-load headCount from localStorage', () => {
    render(<ComplianceCalculator />);
    
    fireEvent.change(screen.getByLabelText(/年度总预算/i), { target: { value: '100000' } });
    
    // 人均 = 100000 / 100 = 1000
    expect(screen.getByText(/1000 元/i)).toBeInTheDocument();
  });
});
```

## P2: 三方询价单自动化测试

### QuotationGenerator.test.ts

```typescript
import { generateQuotationSheet } from '../src/utils/quotationGenerator';

describe('QuotationGenerator - P2 Tests', () => {
  const baseAmount = 10000;

  // TC-P2-001: 询价单生成
  test('should generate quotation with supplier A matching base amount', () => {
    const quotation = generateQuotationSheet(baseAmount);
    
    expect(quotation.supplierA).toBe(baseAmount);
  });

  // TC-P2-002: 随机范围验证
  test('should generate supplier B and C within ±5% range', () => {
    const results = [];
    
    // 生成100次验证随机范围
    for (let i = 0; i < 100; i++) {
      const quotation = generateQuotationSheet(baseAmount);
      results.push({
        b: quotation.supplierB,
        c: quotation.supplierC,
      });
    }
    
    const minPrice = baseAmount * 0.95; // 9500
    const maxPrice = baseAmount * 1.05; // 10500
    
    results.forEach(({ b, c }) => {
      expect(b).toBeGreaterThanOrEqual(minPrice);
      expect(b).toBeLessThanOrEqual(maxPrice);
      expect(c).toBeGreaterThanOrEqual(minPrice);
      expect(c).toBeLessThanOrEqual(maxPrice);
    });
  });

  test('should mark B and C as sample data', () => {
    const quotation = generateQuotationSheet(baseAmount);
    
    expect(quotation.notes).toContain('示例数据');
  });
});
```

## P3: 采购地区、节日类型自动化测试

```jsx
import { render, screen, fireEvent } from '@testing-library/react';
import BasicInfoForm from '../src/components/BasicInfoForm';

describe('BasicInfoForm - P3 Tests', () => {
  const mockOnChange = jest.fn();

  // TC-P3-001: 采购地区字段显示
  test('should display region field with default value and readonly', () => {
    render(<BasicInfoForm formData={{}} onChange={mockOnChange} />);
    
    const regionInput = screen.getByLabelText(/采购地区/i);
    expect(regionInput).toHaveValue('新疆地区');
    expect(regionInput).toBeDisabled();
  });

  // TC-P3-002: 节日类型联动-传统节日慰问
  test('should show festival dropdown when scene is traditional festival', () => {
    render(<BasicInfoForm formData={{ scene: 'traditional' }} onChange={mockOnChange} />);
    
    expect(screen.getByLabelText(/节日类型/i)).toBeInTheDocument();
    expect(screen.getByText(/春节/i)).toBeInTheDocument();
    expect(screen.getByText(/古尔邦节/i)).toBeInTheDocument();
    expect(screen.getByText(/肉孜节/i)).toBeInTheDocument();
  });

  // TC-P3-003: 节日类型联动-专项活动物资
  test('should hide festival field when scene is activity', () => {
    render(<BasicInfoForm formData={{ scene: 'activity' }} onChange={mockOnChange} />);
    
    expect(screen.queryByLabelText(/节日类型/i)).not.toBeInTheDocument();
  });

  // TC-P3-004: 节日类型联动-精准帮扶慰问
  test('should hide festival field when scene is assistance', () => {
    render(<BasicInfoForm formData={{ scene: 'assistance' }} onChange={mockOnChange} />);
    
    expect(screen.queryByLabelText(/节日类型/i)).not.toBeInTheDocument();
  });
});
```

## P4: 年度累计提示自动化测试

### AnnualProgressAlert.test.jsx

```jsx
import { render, screen } from '@testing-library/react';
import AnnualProgressAlert from '../src/components/AnnualProgressAlert';

describe('AnnualProgressAlert - P4 Tests', () => {
  // TC-P4-001: 首次采购提示
  test('should show first purchase message when no history', () => {
    render(<AnnualProgressAlert history={[]} currentBudget={50000} />);
    
    expect(screen.getByText(/本年度首次采购/i)).toBeInTheDocument();
  });

  // TC-P4-002: 累计进度显示
  test('should display completion percentage based on history', () => {
    const history = [
      { totalAmount: 50000 },
      { totalAmount: 30000 },
    ];
    render(<AnnualProgressAlert history={history} currentBudget={100000} />);
    
    // 已完成8万，当前预算10万，完成率80%
    expect(screen.getByText(/本年度已完成 80%/i)).toBeInTheDocument();
  });

  // TC-P4-004: 超标提醒
  test('should show warning when exceeding 2000 yuan per capita', () => {
    const history = [{ totalAmount: 180000 }]; // 已超18万
    render(<AnnualProgressAlert history={history} currentBudget={50000} headCount={100} />);
    
    // 累计23万/100人 = 2300元/人 > 2000元
    expect(screen.getByText(/年度人均超标/i)).toBeInTheDocument();
    expect(screen.getByText(/2300 元.*超过 2000 元/i)).toBeInTheDocument();
  });
});
```

## 端到端测试 (E2E)

### complete-workflow.test.js (使用Playwright)

```javascript
const { test, expect } = require('@playwright/test');

test.describe('Complete Workflow - E2E', () => {
  test('TC-E2E-001: full procurement workflow', async ({ page }) => {
    // Step 1: Page1 - Fill basic info
    await page.goto('http://localhost:5173');
    await page.fill('[name="headCount"]', '100');
    await page.fill('[name="totalBudget"]', '80000');
    await page.selectOption('[name="scene"]', 'traditional');
    await page.selectOption('[name="festival"]', 'spring');
    
    // Verify no warning for 800 yuan per capita
    await expect(page.locator('.warning-500')).not.toBeVisible();
    
    // Step 2: Page2 - Generate solution
    await page.click('button:has-text("生成方案")');
    await expect(page.locator('.solution-preview')).toBeVisible();
    
    // Verify annual progress hint
    await expect(page.locator('.annual-hint')).toContainText('本年度');
    
    // Step 3: Page3 - Generate report
    await page.click('button:has-text("下一步")');
    await page.fill('[name="unitName"]', '测试单位');
    await page.click('button:has-text("生成正式公文")');
    
    // Verify policy reference
    await expect(page.locator('.report-body')).toContainText('新财购〔2025〕2号');
    
    // Step 4: Page4 - Check compliance
    await page.click('button:has-text("合规测算")');
    await page.fill('[name="annualBudget"]', '200000');
    
    // Verify 2000 yuan threshold (no warning at exactly 2000)
    await expect(page.locator('.warning-2000')).not.toBeVisible();
    
    // Change to exceed threshold
    await page.fill('[name="annualBudget"]', '250000');
    await expect(page.locator('.warning-2000')).toBeVisible();
  });
});
```
