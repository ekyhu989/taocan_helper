### 2.2 目录设计原则

**1. 按功能模块划分**：
- `templates/` - 公文模板模块
- `policies/` - 政策文件模块
- 每个模块包含：data（数据）、components（组件）、utils（工具）

**2. 组件分类清晰**：
- `common/` - 通用组件（双端共用）
- `desktop/` - 桌面端专用
- `mobile/` - 移动端专用

**3. 工具函数分类**：
- `algorithm/` - 核心算法
- `compliance/` - 合规计算
- `helpers/` - 辅助函数

```typescript
// src/stores/uiStore.ts
import { create } from 'zustand';

interface Toast {
  id: string;
  type: 'success' | 'warning' | 'error' | 'info';
  message: string;
  description?: string;
}

interface UIState {
  toasts: Toast[];
  isLoading: boolean;
  
  // Actions
  addToast: (toast: Omit<Toast, 'id'>) => void;
  removeToast: (id: string) => void;
  setLoading: (isLoading: boolean) => void;
}

export const useUIStore = create<UIState>((set) => ({
  toasts: [],
  isLoading: false,
  
  addToast: (toast) => {
    const id = Math.random().toString(36).substr(2, 9);
    set((state) => ({
      toasts: [...state.toasts, { ...toast, id }]
    }));
    
    // 自动移除（error类型除外）
    if (toast.type !== 'error') {
      setTimeout(() => {
        set((state) => ({
          toasts: state.toasts.filter((t) => t.id !== id)
        }));
      }, toast.type === 'success' ? 2000 : 3000);
    }
  },
  
  removeToast: (id) =>
    set((state) => ({
      toasts: state.toasts.filter((t) => t.id !== id)
    })),
  
  setLoading: (isLoading) => set({ isLoading }),
}));
```

```typescript
import React, { useState } from 'react';
import { Template, TemplateCategory } from '@/types/template';
import { Button } from '@/components/common/Button';
import { Input } from '@/components/common/Input';
import { Card } from '@/components/common/Card';

interface TemplateSelectorProps {
  templates: Template[];
  onSelect: (template: Template) => void;
  onClose: () => void;
}

const TemplateSelector: React.FC<TemplateSelectorProps> = ({
  templates,
  onSelect,
  onClose
}) => {
  const [searchTerm, setSearchTerm] = useState('');
  const [selectedCategory, setSelectedCategory] = useState<TemplateCategory | null>(null);

  // 过滤模板
  const filteredTemplates = templates.filter((template) => {
    const matchSearch = template.name.toLowerCase().includes(searchTerm.toLowerCase()) ||
                       template.description.toLowerCase().includes(searchTerm.toLowerCase());
    const matchCategory = selectedCategory ? template.category === selectedCategory : true;
    return matchSearch && matchCategory;
  });

  return (
    <div className="template-selector">
      <div className="template-selector__header">
        <h2>选择公文模板</h2>
        <button onClick={onClose} aria-label="关闭">×</button>
      </div>
      
      <div className="template-selector__filters">
        <Input
          type="text"

        />
        
        <div className="template-selector__categories">
```

```typescript
// src/templates/utils/variableReplacer.ts
import { TemplateVariable } from '@/types/template';

interface VariableValues {
  [key: string]: string;
}

/**
 * 替换模板中的变量
 * @param content 模板内容
 * @param variables 变量值映射
 * @returns 替换后的内容
 */
export function replaceVariables(content: string, variables: VariableValues): string {
  let result = content;
  
  Object.entries(variables).forEach(([key, value]) => {
    const regex = new RegExp(`\\$\\{${key}\\}`, 'g');
    result = result.replace(regex, value);
  });
  
  return result;
}

/**
 * 验证变量是否完整
 * @param template 模板对象
 * @param values 变量值
 * @returns 是否完整
 */
export function validateVariables(
  template: { variables: TemplateVariable[] },
  values: VariableValues
): { isValid: boolean; missingFields: string[] } {
  const missingFields: string[] = [];
  
  template.variables.forEach((variable) => {
    if (variable.required && !values[variable.key]) {
      missingFields.push(variable.label);
    }
  });
  
  return {
    isValid: missingFields.length === 0,
    missingFields
  };
}
```

#### 搜索引擎

```typescript
// src/policies/utils/searchEngine.ts
import { Policy } from '@/types/policy';

interface SearchOptions {
  searchTerm?: string;
  category?: string;
  scene?: string;
  year?: number;
  level?: string;
}

/**
 * 政策搜索
 * @param policies 政策列表
 * @param options 搜索选项
 * @returns 过滤后的政策列表
 */
export function searchPolicies(
  policies: Policy[],
  options: SearchOptions
): Policy[] {
  let result = [...policies];
  
  // 全文搜索
  if (options.searchTerm) {
    const term = options.searchTerm.toLowerCase();
    result = result.filter(
      (policy) =>
        policy.title.toLowerCase().includes(term) ||
        policy.content.toLowerCase().includes(term) ||
        policy.summary.toLowerCase().includes(term)
    );
  }
  
  // 分类筛选
  if (options.category) {
    result = result.filter((policy) => policy.category === options.category);
  }
  
  // 场景筛选
  if (options.scene) {
    result = result.filter((policy) => policy.scene.includes(options.scene as any));
  }
  
  // 年份筛选
  if (options.year) {
    result = result.filter((policy) => policy.year === options.year);
  }
  
  // 等级筛选
  if (options.level) {
    result = result.filter((policy) => policy.level === options.level);
  }
  
  return result;
}

/**
 * 高亮搜索关键词
 * @param text 原文本
 * @term 关键词
 * @returns 高亮后的HTML
 */
export function highlightKeyword(text: string, term: string): string {
  if (!term) return text;
  
  const regex = new RegExp(`(${term})`, 'gi');
  return text.replace(regex, '<mark>$1</mark>');
}
```

### 4.4 缓存策略

#### Service Worker（可选）

```typescript
// public/sw.js
const CACHE_NAME = 'taocang-v3.0';
const ASSETS_TO_CACHE = [
  '/',
  '/index.html',
  '/assets/vendor-[hash].js',
  '/assets/main-[hash].js',
  '/assets/main-[hash].css'
];

self.addEventListener('install', (event) => {
  event.waitUntil(
    caches.open(CACHE_NAME).then((cache) => cache.addAll(ASSETS_TO_CACHE))
  );
});

self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((response) => {
      return response || fetch(event.request);
    })
  );
});
```

## 📝 九、技术决策记录

### 9.1 为什么选择HashRouter？

**决策**：使用HashRouter替代BrowserRouter

**原因**：
1. 微信内置浏览器对SPA路由重写兼容性差
2. Hash路由无需服务器配置
3. 深链接访问稳定可靠

**权衡**：
- ✅ 优点：微信兼容性好
- ❌ 缺点：URL不美观（带#号）
- ❌ 缺点：不支持SEO（但本工具不需要）

### 9.2 为什么保持Zustand？

**决策**：继续使用Zustand，不引入Redux

**原因**：
1. 项目规模适中，不需要Redux的复杂性
2. Zustand API简洁，学习成本低
3. 性能优秀，支持persist中间件

### 9.3 为什么选择Lucide Icons？

**决策**：使用Lucide Icons替换Emoji

**原因**：
1. 线性简洁风格，符合政企专业感
2. 支持tree-shaking，减小体积
3. MIT许可，商业友好
4. 持续维护，图标库完整
